def dividir(a, b):
    if b == 0:
        return "Erro: divisão por zero!"
    return a / b
 
while True:
    opcao = input("Escolha uma opção: ")
    if opcao == "0": break
    a = float(input("Primeiro número: "))
    b = float(input("Segundo número: "))
    print("Resultado:", dividir(a, b))
